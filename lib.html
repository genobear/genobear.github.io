import time
import os
import requests
from datetime import datetime, timedelta
import traceback

DATE_FORMAT = '%Y-%m-%d'
DATE_STEP = timedelta(days=1)


def _strptime(string):
    return datetime.strptime(string, DATE_FORMAT)


def _strftime(date):
    return date.strftime(DATE_FORMAT)


def _date_range_parameters(start, end, span_days):
    start = _strptime(start)
    end = _strptime(end)
    span = timedelta(days=span_days)
    return start, end, span


def forward_date_range(start: str, end: str, span_days: int) -> tuple:
    """_Generate tuples of strings with intervals from given range of dates (forward)_

    Args:
        start (str): _YYYY-MM-DD_
        end (str): _YYYY-MM-DD_
        span_days (int): _Span of days_

    Returns:
        tuple: _A tuple including two dates formatted as strings 'YYYY-MM-DD'_

    Yields:
        Iterator[tuple]: _Example: backward_date_range('2012-01-01', '2012-01-05', 2)_  
            _1st yield = ('2012-01-01'', '2012-01-03')_  
            _2nd yield = ('2012-01-04', '2012-01-05')_
    """
    start, end, span = _date_range_parameters(start, end, span_days)
    stop = end - span

    while start < stop:
        current = start + span
        yield _strftime(start), _strftime(current)
        start = current + DATE_STEP

    yield str(_strftime(start)), str(_strftime(end))


def backward_date_range(start: str, end: str, span_days: int) -> tuple:
    """_Generate tuples of strings with intervals from given range of dates (backward)_

    Args:
        start (str): _YYYY-MM-DD_
        end (str): _YYYY-MM-DD_
        span_days (int): _Span of days_

    Returns:
        tuple: _A tuple including two dates formatted as strings 'YYYY-MM-DD'_

    Yields:
        Iterator[tuple]: _backward_date_range('2012-01-01', '2012-01-05', 2)_  
            _1st yield = ('2012-01-04'', '2012-01-05')_  
            _2nd yield = ('2012-01-01', '2012-01-03')_
    """
    start, end, span = _date_range_parameters(start, end, span_days)
    stop = start + span

    while end > stop:
        current = end - span
        yield _strftime(current), _strftime(end)
        end = current - DATE_STEP

    yield str(_strftime(start)), str(_strftime(end))


def dr_retry_check(content: str) -> bool:
    """_Checks string for specific errors from the DR service. 
    Errors can be returned even though response code is 200(succesfully responded with error)_. 

    Args:
        content (str): _response.text from requester_

    Raises:
        ValueError: _Could not log in_
        ValueError: _Configuration error_
        ValueError: _E4042_APPDN_NOT_FOUND_

    Returns:
        bool: _True if soft error(timeout, rate limt)_  
            _False if Download Succesful_ 
    """
    if "Failed to Connect" in content:
        print("Failed to connect, trying again...")
        return True
    elif "ACCESS DENIED" in content:
        print("ACCESS DENIED - waiting 60s")
        time.sleep(60)
        return True
    elif "Could not log in" in content:
        raise ValueError("Could not log in")
    elif "Configuration error" in content:
        raise ValueError("Configuration error")
    elif "E4042_APPDN_NOT_FOUND" in content:
        print("E4042_APPDN_NOT_FOUND")
        raise ValueError("E4042_APPDN_NOT_FOUND")
    else:
        print("Dowload Succesful")
        return False


def download_archipel(start_date: str, end_date: str, login: str, password: str, acceptor: str):
    """
    _Download archipel data from data requester_  
    _Start date always starts from 00:00:00_  
    _End date always ends at 23:59:59_  

    Args:
        start_date (str): _'YYYY-MM-DD'_ 
        end_date (str): _'YYYY-MM-DD'_
        login (str): _DR login from DC_
        password (str): _DR password from DC_
        acceptor (str): _Acceptor 'cn' name_

    Returns:
        str: _returns response.text from requester, usually in csv format_
    """
    # reformat date into datetime for DR
    startdatetime = datetime.strptime(start_date, DATE_FORMAT)
    startdatetime = datetime.strftime(startdatetime, '%Y%m%d')
    startdatetime = f'{startdatetime}000000'
    enddatetime = datetime.strptime(end_date, DATE_FORMAT)
    enddatetime = datetime.strftime(enddatetime, '%Y%m%d')
    enddatetime = f'{enddatetime}235959'

    status = True

    while status:
        payload = {
            'login': login,
            'password': password,
            'city': acceptor,
            'report': 'archipel_transactionspub',
            'startdatetime': startdatetime,
            'enddatetime': enddatetime
        }
        print(
            f"Start DL for {acceptor}/nStart:{startdatetime} End:{enddatetime}")
        response = requests.post(
            'https://my.flowbird.io/drdirect.php', data=payload)
        print(response)
        status = dr_retry_check(response.text)
        if status:
            pass
        else:
            return response.text
