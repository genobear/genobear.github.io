import lib as DR
import traceback

def download_archipel_from_list(login, password, start, end):
    """Opens acceptors.txt from root directory. Acceptor must be the 'cn' name.
    Iterates through acceptors and saves data to 'data/archipel'

    Args:
        login (string): Datarequester login from DC
        password (string): Datarequester password from DC
        start (string): Start date and time YYYYMMDDHHMMSS e.g'20220130000000'
        end (string): End date and time YYYYMMDDHHMMSS e.g'20220130000000'
    """
    if not os.path.exists(f'data/archipel'):
        os.makedirs(f'data/archipel')

    filename = f"acceptors.txt"
    with open(filename, "r") as list:

        for acceptor in list:
            acceptor = acceptor.rstrip('\n')
            print(acceptor)
            status = True
            while status:
                payload = {
                    'login': login,
                    'password': password,
                    'city': acceptor,
                    'report': 'archipel_transactionspub',
                    'startdatetime': start,
                    'enddatetime': end
                }
                print(f"Start DL for {acceptor}")
                response = requests.post(
                    'https://my.flowbird.io/drdirect.php', data=payload)
                print(response)
                status = DR.dr_retry_check(response.text)
                if status:
                    pass
                else:
                    with open(f"data/archipel/{acceptor}_{start}_{end}.csv", "w", encoding="utf-8") as f:
                        f.write(response.text)
        print("All Archipel DL Finished")
    pass

def batch_download_archipel(start_date, end_date, login, password, acceptor, batch_size):
    """Download large data set in smaller batches. Saves each batch in seperate file. 
    Batches are saved to 'data/archipel/{acceptor}'

    Args:
        start_date (string): YYYY-MM-DD e.g 2022-01-30 the start time is always 00:00:00
        end_date (string): YYYY-MM-DD e.g 2022-01-30 the end time is always 23:59:59
        login (string): DR login from DC
        password (string): DR login from DC
        acceptor (string): Acceptor 'cn' name
        batch_size (int): Number of days in each batch
    """
    if not os.path.exists(f'data/archipel/{acceptor}'):
        os.makedirs(f'data/archipel/{acceptor}')

    for start, end in DR.forward_date_range(start_date, end_date, batch_size):
        print('')
        data = DR.download_archipel(start, end, login, password, acceptor)
        with open(f"data/archipel/{acceptor}/{acceptor}_{start}_{end}.csv", "w", encoding="utf-8") as f:
            f.write(data)
    pass
